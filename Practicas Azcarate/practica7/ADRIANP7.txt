CREATE TABLE USUARIO
(
    COD VARCHAR2(10)NOT NULL
    NOMBRE VARCHAR2(15)NOT NULL
    APELLIDO VARCHAR2(55)NOT NULL
    EDAD NUMBER(2) 
    SEXO VARCHAR2(1) NOT NULL
    CONSTRAINT PK_USUARIO PRIMARY KEY(COD)
)
CREATE TABLE LIBRO
(
    ISBN VARCHAR2(10)
    CLASE VARCHAR2(1)
    TITULO VARCHAR2(35)
    AUTOR VARCHAR2(15)
    EDITORIAL VARCHAR2(25)
    CONSTRAINT PK_LIBRO PRIMARY KEY (ISBN)
)
CREATE TABLE PRESTAMO
(
    ISBN VARCHAR2(10)
    COD VARCHAR2(10)
    FECHA_INI DATE
    FECHA_FIN DATE
    CONSTRAINT PK_PRESTAMO PRIMARY KEY (ISBN)
    CONSTRAINT FK_PRESTAMO FOREIGN KEY (COD) REFERENCES USUARIO(COD)
)
CREATE TABLE CLASELIBRO
(
    CLASE VARCHAR2(1)
    TIEMPO NUMBER(5)
    CONSTRAINT PK_CLASELIBRO PRIMARY KEY(CLASE)
)
